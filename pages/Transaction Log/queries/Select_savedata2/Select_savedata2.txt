  date, jumlahbottle, saldo, transactioncode
FROM
  savedata
WHERE
  transactioncode LIKE '%{{table_savedata.searchText}}%'
ORDER BY
  {{table_savedata.sortOrder.column || 'no'}} {{table_savedata.sortOrder.order !== "desc" ? "" : "DESC"}}
LIMIT
  {{table_savedata.pageSize}}
OFFSET
  {{table_savedata.pageOffset}}
